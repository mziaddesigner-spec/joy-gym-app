function doPost(e) { try { // 1. استقبال البيانات وتفكيكها var data = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); // 2. معالجة الصورة وتحويلها من Base64 إلى ملف حقيقي var contentType = data.image.split(",")[0].split(":")[1].split(";")[0]; var decode = Utilities.base64Decode(data.image.split(",")[1]); var fileName = data.name + "_" + new Date().getTime() + ".jpg"; var blob = Utilities.newBlob(decode, contentType, fileName); // 3. حفظ الملف في المجلد المحدد var folderId = "ضع_هنا_معرف_المجلد_الخاص_بك"; // استبدل هذا الجزء var folder = DriveApp.getFolderById(folderId); var file = folder.createFile(blob); // جعل الملف متاحاً للعرض عبر الرابط file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); var fileUrl = file.getUrl(); // 4. تحديد الرقم التسلسلي (ID) var lastRow = sheet.getLastRow(); var nextId = lastRow < 2 ? 101 : parseInt(sheet.getRange(lastRow, 1).getValue()) + 1; // 5. تسجيل البيانات في الجدول حسب الترتيب المطلوب sheet.appendRow([ nextId, // العمود A: المعرف data.name, // العمود B: الاسم "", // العمود C: (صورة المشترك - اتركها فارغة) fileUrl, // العمود D: رابط السحاب "", // العمود E: (رمز QR - يمكن إضافته لاحقاً) new Date() // العمود F: التاريخ ]); return ContentService.createTextOutput(JSON.stringify({ "result": "success", "id": nextId })).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ "result": "error", "message": error.toString() })).setMimeType(ContentService.MimeType.JSON); } }
