    <div class="upload-zone" onclick="document.getElementById('imageInput').click()">
        <span>📷 التقط أو اختر صورة الشخصية</span>
        <input type="file" id="imageInput" accept="image/*" style="display:none">
    </div>

    <div id="upload-demo"></div>
    
    <button id="saveBtn" class="btn-save" onclick="processRegistration()">إصدار الهوية وحفظ البيانات</button>

<script src="https://cdnjs.cloudflare.com/ajax/libs/croppie/2.6.5/croppie.min.js"></script>

<script>
    // ==========================================
    // إعدادات الربط - استبدل القيم هنا بدقة
    // ==========================================
    const CLOUD_NAME = "mediaflows_63e9d71e-1d55-4d86-a638-15769a0309b4";
    const UPLOAD_PRESET = "joy_gym_preset"; 
    const APPS_SCRIPT_URL = "ضع_رابط_الـ_EXEC_هنا"; 

    // إعداد أداة القص
    let croppieInstance = new Croppie(document.getElementById('upload-demo'), {
        viewport: { width: 220, height: 220, type: 'circle' },
        boundary: { width: 300, height: 300 },
        showZoomer: true,
        enableOrientation: true
    });

    // معالجة اختيار الصورة
    document.getElementById('imageInput').onchange = function() {
        const reader = new FileReader();
        reader.onload = function(e) {
            document.getElementById('upload-demo').style.display = 'block';
            document.getElementById('saveBtn').style.display = 'block';
            croppieInstance.bind({ url: e.target.result });
        }
        reader.readAsDataURL(this.files[0]);
    };

    // الوظيفة الأساسية
    async function processRegistration() {
        const name = document.getElementById('userName').value;
        const status = document.getElementById('status');
        const saveBtn = document.getElementById('saveBtn');

        if (!name) { alert("من فضلك أدخل اسم المشترك أولاً"); return; }
        
        saveBtn.disabled = true;
        status.innerHTML = "⏳ جاري ضغط الصورة ورفعها لسحابة Cloudinary...";

        try {
            // 1. استخراج الصورة من أداة القص بصيغة Blob
            const blob = await croppieInstance.result({
                type: 'blob',
                size: { width: 400, height: 400 },
                format: 'jpeg',
                circle: false // نأخذها مربعة لسهولة العرض في الجدول
            });

            // 2. الرفع لـ Cloudinary
            const formData = new FormData();
            formData.append('file', blob);
            formData.append('upload_preset', UPLOAD_PRESET);

            const cloudRes = await fetch(`https://api.cloudinary.com/v1_1/${CLOUD_NAME}/image/upload`, {
                method: 'POST',
                body: formData
            });
            const cloudData = await cloudRes.json();
            
            if (!cloudData.secure_url) throw new Error("فشل الرفع لمستودع الصور");
            const imageUrl = cloudData.secure_url;

            status.innerHTML = "✅ تم رفع الصورة بنجاح!<br>⏳ جاري إرسال البيانات لجدول جوجل...";

            // 3. إرسال البيانات للجسر (App Script)
            // ملاحظة: نستخدم mode: 'no-cors' لتجنب مشاكل المتصفح الأمنية
            await fetch(APPS_SCRIPT_URL, {
                method: 'POST',
                mode: 'no-cors',
                cache: 'no-cache',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    name: name,
                    imageUrl: imageUrl
                })
            });

            status.innerHTML = "<span class='success-msg'>🎉 مبروك! تم تسجيل المشترك وحفظ بياناته بنجاح في الجدول.</span>";
            alert("تمت العملية بنجاح!");
            
            // إعادة تعيين الواجهة
            setTimeout(() => location.reload(), 3000);

        } catch (error) {
            console.error(error);
            status.innerHTML = "❌ حدث خطأ: " + error.message;
            saveBtn.disabled = false;
        }
    }
</script>
